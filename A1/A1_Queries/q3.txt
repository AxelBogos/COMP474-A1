#Which courses at Concordia teach Knowledge graph?

prefix a1_data: <http://a1.io/data#> 
prefix a1_schema: <http://a1.io/schema#>
prefix dbp: <http://dbpedia.org/property/>
prefix dbr: <http://dbpedia.org/resource/>

SELECT ?Course
WHERE{
  	a1_data:Concordia_University a1_schema:Offers ?Course .
        ?Course a1_schema:HasLecture ?Lecture .
        ?Lecture dbp:subject dbr:Knowledge_graph
    }
